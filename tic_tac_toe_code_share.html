[ Tic Tac Toe — C++ ]
#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

char board[3][3];
char player = 'X';

void initBoard()
{
    char n = '1';
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            board[i][j] = n++;
}

void printBoard()
{
    for (int i = 0; i < 3; i++)
    {
        cout << " ";
        for (int j = 0; j < 3; j++)
        {
            if(board[i][j] == 'X')
                cout << "\033[38;2;255;0;0mX\033[0m";
            else if(board[i][j] == 'O')
                cout << "\033[38;2;0;255;0mO\033[0m";
            else
                cout << board[i][j];
            if(j < 2) cout << " | ";
        }
        cout << endl;
        if(i < 2) cout << "---|---|---" << endl;
    }
}

void playerMove(char player)
{
    char pos;
    while (true)
    {
        cout << "\nPlayer " << player << ", enter position(1-9) : ";
        cin >> pos;
        for (int i = 0; i < 3; i++)
            for (int j = 0; j < 3; j++)
                if(board[i][j] == pos)
                { board[i][j] = player; return; }
        cout << "\nInvalid position!" << endl;
    }
}

bool checkWinner(char player)
{
    for (int i = 0; i < 3; i++)
        if(board[i][0]==player && board[i][1]==player && board[i][2]==player)
            return true;
    for (int i = 0; i < 3; i++)
        if(board[0][i]==player && board[1][i]==player && board[2][i]==player)
            return true;
    if(board[0][0]==player && board[1][1]==player && board[2][2]==player)
        return true;
    if(board[0][2]==player && board[1][1]==player && board[2][0]==player)
        return true;
    return false;
}

bool checkTie()
{
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            if(board[i][j] >= '1' && board[i][j] <= '9')
                return false;
    return true;
}

void botMove()
{
    char cells[9]; int count = 0;
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            if(board[i][j] >= '1' && board[i][j] <= '9')
                cells[count++] = board[i][j];
    char move = cells[rand() % count];
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            if(board[i][j] == move)
            { board[i][j] = 'O'; return; }
}

int main()
{
    srand(time(0));
    initBoard();
    while (true)
    {
        system("cls");
        printBoard();
        if(player == 'X') playerMove(player);
        else botMove();

        if(checkWinner(player))
        {
            system("cls"); printBoard();
            if(player == 'X')
                cout << "\n\033[38;2;255;0;0m" << player << " is the winner!\033[0m" << endl;
            else
                cout << "\n\033[38;2;0;255;0m" << player << " is the winner!\033[0m" << endl;
            break;
        }
        if(checkTie())
        {
            system("cls"); printBoard();
            cout << "\n\033[38;2;255;255;0mIt's tie!\033[0m" << endl;
            break;
        }
        player = (player == 'X') ? 'O' : 'X';
    }
    return 0;
}
